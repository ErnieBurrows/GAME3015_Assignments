96 static UINT CalcConstantBufferByteSize(UINT byteSize)
97 {
98 // Constant buffers must be a multiple of the minimum hardware
99 // allocation size (usually 256 bytes). So round up to nearest
100 // multiple of 256. We do this by adding 255 and then masking off
101 // the lower 2 bytes which store all bits < 256.
102 // Example: Suppose byteSize = 300.
103 // (300 + 255) & ~255
104 // 555 & ~255
105 // 0x022B & ~0x00ff
106 // 0x022B & 0xff00
107 // 0x0200
108 // 512
109 return (byteSize + 255) & ~255;
110 }
